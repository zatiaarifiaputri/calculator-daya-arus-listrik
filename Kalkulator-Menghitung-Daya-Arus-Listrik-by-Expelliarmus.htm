# Fungsi untuk menghitung daya def hitung_daya(arus, tegangan): daya = arus * tegangan return daya # Fungsi untuk menghitung arus def hitung_arus(daya, tegangan): arus = daya / tegangan return arus # Fungsi untuk menghitung tegangan def hitung_tegangan(daya, arus): tegangan = daya / arus return tegangan # Program utama def main(): print("Kalkulator Daya dan Arus Listrik by Expelliarmus") print("Pilih operasi yang ingin Anda lakukan:") print("1. Hitung Daya(W)") print("2. Hitung Arus(A)") print("3. Hitung Tegangan(V)") pilihan = input("Masukkan pilihan (1/2/3): ") if pilihan == '1': print("Anda Memilih Hitung Daya(W) dengan rumus Arus(A) dikali Tegangan(V)") arus = float(input("Masukkan nilai arus (A): ")) tegangan = float(input("Masukkan nilai tegangan (V): ")) print("Daya adalah:", hitung_daya(arus, tegangan), "Watt") elif pilihan == '2': print("Anda Memilih Hitung Arus(A) dengan rumus Daya(W) dibagi Tegangan(V)") daya = float(input("Masukkan nilai daya (W): ")) tegangan = float(input("Masukkan nilai tegangan (V): ")) print("Arus adalah:", hitung_arus(daya, tegangan), "Ampere") elif pilihan == '3': print("Anda Memilih Hitung Tegangan dengan rumus Daya(W) dibagi Arus(A)") daya = float(input("Masukkan nilai daya (W): ")) arus = float(input("Masukkan nilai arus (A): ")) print("Tegangan adalah:", hitung_tegangan(daya, arus), "Volt") else: print("Pilihan tidak valid.") if __name__ == "__main__": main()
